radiobutton 单选框

概述
  • <input type="radio" ng-model="string" value="string" [name="string"] [ng-change="string"] ng-value="string">
  • 如果想绑定非字符串的值，用ng-value代替value属性

https://docs.angularjs.org/api/ng/input/input%5Bradio%5D

实例

从数据字典读取选择项，绑定值

vm.classOnePick: {{vm.classOnePick}}

代码 
<form name="form1" class="form-horizontal" novalidate>
<div class="form-group">
  <label class="control-label col-md-2">radio1:</label>
  <div class="col-md-4">
    <label ng-repeat="cls in classifications.classOne" class="radio-inline">
      <input name="radio1" type="radio" ng-model="vm.classOnePick" ng-value="cls.id" ng-required="true">{{cls.name}}
    </label>
    <div class="text-danger" ng-messages="form1.radio1.$error">
      <div ng-messages-include="/components/tpls/errorMessages.html"></div>
    </div>
  </div>
  <div class="col-md-6">
    <p class="form-control-static">vm.classOnePick: {{vm.classOnePick}}</p>
  </div>
</div>
</form>

$scope.classifications = {
  classOne: [{
    id: 1,
    name: '选项1',
    value: 'value1'
  }, {
    id: 2,
    name: '选项2',
    value: 'value2'
  }, {
    id: 3,
    name: '选项3',
    value: 'value3'
  }]
};

$scope.vm = {
  classOnePick : null
};
