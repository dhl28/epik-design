datepicker 日期选择框

概述
  • 日期选择框使用ui.bootstrap.datepicker
  • 使用angular-i18n做了本地化配置
  • 支持选择年，月，日
  • 日期返回格式均为date类型
  • 为兼容metronic,angular-bootstrap版本0.13.0
  • 全局配置
    • datepickerConfig.minDate = new Date(1900, 1, 1);
    • datepickerConfig.maxDate = new Date(2100, 1, 1);
    • datepickerConfig.formatData = 'dd';
    • datepickerConfig.formatMonth = 'MMM';
    • datepickerConfig.formatYear = 'yyyy';
    • datepickerConfig.formatDayHeader = 'EEE';
    • datepickerConfig.formatDayTitle = 'yyyy-MM';
    • datepickerPopupConfig.currentText = '今天';
    • datepickerPopupConfig.clearText = '清除';
    • datepickerPopupConfig.closeText = '关闭';
实例

date1 : {{vm.date1}}

date2 : {{vm.date2}}

代码 
<form name="form1" class="form-horizontal" novalidate>
<div class="form-group" ng-class="{ 'has-error' : form1.date1.$invalid && !form1.date1.$pristine }">
<label class="control-label col-md-2" for="date1">date1:</label>
<div class="col-md-4">
<p class="input-group">
  <input type="text" name="date1" id="date1" class="form-control" datepicker-popup="yyyy-MM"
         datepicker-options="datepickerCtrl.datePickerOptions1" ng-model="vm.date1"
         is-open="datepickerCtrl.date1Opened" ng-required="true" ng-disabled="true"/>
    <span class="input-group-btn">
      <button type="button" class="btn btn-default"
              ng-click="datepickerCtrl.funOpenDate1($event)"><i
        class="glyphicon glyphicon-calendar"></i></button>
    </span>
</p>
<div class="help-block" ng-messages="form1.date1.$error" ng-show="form1.date2.$dirty">
  <div ng-messages-include="/components/tpls/errorMessages.html"></div>
</div>
</div>
<div class="col-md-6">
<p class="form-control-static">date1 : {{vm.date1}}</p>
</div>
</div>
<div class="form-group" ng-class="{ 'has-error' : form1.date2.$invalid && !form1.date2.$pristine }">
<label class="control-label col-md-2" for="date2">date2:</label>
<div class="col-md-4">
<p class="input-group">
  <input type="text" name="date2" id="date2" class="form-control" datepicker-popup
         ng-model="vm.date2" is-open="datepickerCtrl.date2Opened" ng-required="true"
         ng-disabled="true"/>
    <span class="input-group-btn">
      <button type="button" class="btn btn-default"
              ng-click="datepickerCtrl.funOpenDate2($event)"><i
        class="glyphicon glyphicon-calendar"></i></button>
    </span>
</p>
<div class="help-block" ng-messages="form1.date2.$error" ng-show="form1.date2.$dirty">
  <div ng-messages-include="/components/tpls/errorMessages.html"></div>
</div>
</div>
<div class="col-md-6">
<p class="form-control-static">date2 : {{vm.date2}}</p>
</div>
</div>

$scope.vm = {
  date1 : new Date(),
  date2 : null
}

$scope.datepickerCtrl = {
  date1Opened : false,
  funOpenDate1 : function($event) {
    $event.preventDefault();
    $event.stopPropagation();
    $scope.datepickerCtrl.date1Opened = true;
  },
  datePickerOptions1: {
    'datepicker-mode': "'month'",
    'min-mode': 'month'
  },
  date2Opened : false,
  funOpenDate2 : function($event) {
    $event.preventDefault();
    $event.stopPropagation();
    $scope.datepickerCtrl.date2Opened = true;
  }
}
